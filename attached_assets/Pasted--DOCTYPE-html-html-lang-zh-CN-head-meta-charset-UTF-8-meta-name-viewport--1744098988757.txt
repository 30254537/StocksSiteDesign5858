    <script>
        // 从 URL 获取产品 ID
        const urlParams = new URLSearchParams(window.location.search);
        const productId = urlParams.get('id');

        // 加载产品详情
        async function loadProduct() {
            try {
                const response = await fetch(`http://localhost:3000/api/products`);
                const products = await response.json();
                const product = products.find(p => p.id == productId);
                if (product) {
                    document.getElementById('product-image').src = `http://localhost:3000${product.imagePath}`;
                    document.getElementById('product-image').alt = product.name;
                    document.getElementById('product-name').textContent = product.name;
                    document.getElementById('product-description').textContent = product.description || '暂无描述';
                    document.getElementById('product-price').textContent = `${product.price} $STONKS`;
                    document.getElementById('product-stock').textContent = product.stock > 0 ? `库存：${product.stock}` : '已售罄';
                    const addToCartBtn = document.getElementById('add-to-cart');
                    if (product.stock > 0) {
                        addToCartBtn.disabled = false;
                        addToCartBtn.onclick = () => addToCart(product);
                    }
                } else {
                    window.location.href = '404.html';
                }
            } catch (error) {
                alert('无法加载产品详情，请确保后端已启动！');
            }
        }

        // 添加到购物车（占位，需后端支持）
        function addToCart(product) {
            alert(`已将 ${product.name} 添加到购物车！`);
            // 实际应用中需保存到本地存储或后端
        }

        loadProduct();
    </script>
