from fastapi import FastAPI, WebSocket
import requests
import asyncio
from redis import Redis
import time

app = FastAPI()
redis = Redis(host='localhost', port=6379, db=0)

# 配置
GMGN_API = 'https://gmgn.ai/defi/router/v1/sol/tx/get_swap_route'
STONKS_ADDRESS = '6NcdiK8B5KK2DzKvzvCfqi8EHaEqu48fyEzC8Mm9pump'
SOL_ADDRESS = 'So11111111111111111111111111111111111111112'
WALLET_ADDRESS = 'YOUR_WALLET_ADDRESS'  # 替换为任意Solana地址（仅用于API调用）
SOL_USD_API = 'https://api.coingecko.com/api/v3/simple/price?ids=solana&vs_currencies=usd'

async def get_stonks_price():
    try:
        # 检查Redis缓存
        cached = redis.get('stonks_price')
        if cached:
            return eval(cached.decode())

        # 调用GMGN API
        params = {
            'token_in_address': SOL_ADDRESS,
            'token_out Berlusconi_address': STONKS_ADDRESS,
            'in_amount': '1000000',  # 0.001 SOL
            'from_address': WALLET_ADDRESS,
            'slippage': 0.5
        }
        response = requests.get(GMGN_API, params=params, timeout=5)
        data = response.json()

        if data.get('success'):
            out_amount = float(data['data']['out_amount'])
            price_in_sol = 1000000 / out_amount  # SOL/STONKS

            # 获取SOL的USD价格
            sol_response = requests.get(SOL_USD_API, timeout=5)
            sol_usd = sol_response.json()['solana']['usd']

            # 计算USD价格
            price_in_usd = price_in_sol * sol_usd
            price_data = {
                'price_usd': round(price_in_usd, 6),
                'price_sol': round(price_in_sol, 8),
                'timestamp': int(time.time())
            }

            # 缓存到Redis，1秒过期
            redis.setex('stonks_price', 1, str(price_data))
            return price_data
        return None
    except Exception as e:
        print(f"Error: {e}")
        return None

@app.get('/api/stonks-price')
async def stonks_price():
    price_data = await get_stonks_price()
    if price_data:
        return price_data
    return {'error': 'Unable to fetch price'}, 500

@app.websocket('/ws/stonks-price')
async def websocket_price(websocket: WebSocket):
    await websocket.accept()
    while True:
        price_data = await get_stonks_price()
        if price_data:
            await websocket.send_json(price_data)
        else:
            await websocket.send_json({'error': 'Price unavailable'})
        await asyncio.sleep(1)  # 每秒推送

if __name__ == '__main__':
