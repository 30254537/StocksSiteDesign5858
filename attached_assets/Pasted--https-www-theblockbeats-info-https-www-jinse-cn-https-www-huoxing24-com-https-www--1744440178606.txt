步骤3：获取新闻数据
我们将创建两种方法：一种用于 RSS 提要，一种用于网页抓取。
3.1 获取 RSS 源
对于具有 RSS 提要的网站（BlockBeats、PANews、8BTC、Odaily），使用feedparser提取新闻。
代码示例（fetch_rss.py）：
Python
import feedparser

rss_feeds = {
    "blockbeats": "https://www.theblockbeats.info/rss",
    "panews": "https://www.panews.io/rss",
    "8btc": "https://www.8btc.com/rss",
    "odaily": "https://www.odaily.news/rss"
}

def fetch_rss_news():
    news_list = []
    for source, url in rss_feeds.items():
        try:
            feed = feedparser.parse(url)
            for entry in feed.entries[:5]:  # Limit to 5 latest per source
                news_list.append({
                    "source": source,
                    "title": entry.title,
                    "link": entry.link,
                    "published": entry.get("published", ""),
                    "summary": entry.get("summary", ""),
                    "fetched_at": datetime.utcnow().isoformat()
                })
        except Exception as e:
            print(f"Error fetching RSS {source}: {e}")
    return news_list
解释：
将 RSS 提要解析为结构化数据（标题、链接等）。
每个来源限制为 5 个项目以避免超载。
包括用于跟踪新鲜度的fetched_at 。
3.2 网页抓取
对于没有 RSS 的网站（金色财经、火星财经、毕世界、ChainDD、金十），抓取它们的新闻页面。每个网站的 HTML 代码都不同，因此我们需要自定义选择器。
代码示例（fetch_scraped.py）：
Python
import requests
from bs4 import BeautifulSoup
from datetime import datetime

scrape_urls = {
    "jinse": "https://www.jinse.cn/newsflash",
    "huoxing": "https://www.huoxing24.com/news",
    "bishijie": "https://www.bishijie.com/kuaixun",
    "chaindd": "https://www.chaindd.com/news",
    "jin10": "https://www.jin10.com/flash.htm"
}

def fetch_scraped_news():
    news_list = []
    headers = {"User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) Chrome/91.0.4472.124"}
    
    for source, url in scrape_urls.items():
        try:
            response = requests.get(url, headers=headers, timeout=10)
            response.raise_for_status()
            soup = BeautifulSoup(response.text, "html.parser")
            
            # Customize selectors based on site (example for jinse.cn)
            if source == "jinse":
                items = soup.select(".jsx-1657651979.item")[:5]  # Adjust selector
                for item in items:
                    title = item.select_one(".title").text.strip()
                    link = item.select_one("a")["href"]
                    time = item.select_one(".time").text.strip()
                    news_list.append({
                        "source": source,
                        "title": title,
                        "link": "https://www.jinse.cn" + link,
                        "published": time,
                        "summary": "",
                        "fetched_at": datetime.utcnow().isoformat()
                    })
            # Add similar blocks for huoxing, bishijie, chaindd, jin10
            # Example for jin10.com (adjust selector after inspecting HTML)
            elif source == "jin10":
                items = soup.select(".jin-flash_item")[:5]  # Hypothetical selector
                for item in items:
                    title = item.select_one(".jin-flash_title").text.strip()
                    link = item.select_one("a")["href"]
                    time = item.select_one(".jin-flash_time").text.strip()
                    news_list.append({
                        "source": source,
                        "title": title,
                        "link": link,
                        "published": time,
                        "summary": "",
                        "fetched_at": datetime.utcnow().isoformat()
                    })
            # TODO: Implement for huoxing, bishijie, chaindd
        except Exception as e:
            print(f"Error scraping {source}: {e}")
    
    return news_list

步骤4：存储新闻数据
使用MongoDB存储新闻，保证不重复，并能快速检索。
代码示例（database.py）：
Python
from pymongo import MongoClient
from datetime import datetime

client = MongoClient("mongodb://<your-atlas-url>/")
db = client["crypto_news"]
collection = db["news"]

def save_news(news_list):
    for news in news_list:
        collection.update_one(
            {"link": news["link"]},  # Unique identifier
            {"$set": news},
            upsert=True
        )
解释：
update_one与upsert=True添加新新闻或更新现有新闻。
链接确保没有重复，因为新闻 URL 是唯一的。
fetched_at有助于追踪新鲜度。
步骤 5：安排更新
每 5 分钟运行一次获取，以确保新闻接近实时。
代码示例（scheduler.py）：
Python
from apscheduler.schedulers.background import BackgroundScheduler
from fetch_rss import fetch_rss_news
from fetch_scraped import fetch_scraped_news
from database import save_news

def update_news():
    news = fetch_rss_news() + fetch_scraped_news()
    save_news(news)
    print(f"Updated {len(news)} news items at {datetime.utcnow()}")

scheduler = BackgroundScheduler()
scheduler.add_job(update_news, "interval", minutes=5)
scheduler.start()
选择：
使用云调度程序（例如 AWS Lambda、Google Cloud Scheduler）进行生产。

通过 Flask API 公开新闻以供前端使用。
代码示例（app.py）：
Python
from flask import Flask, jsonify
from pymongo import MongoClient
from scheduler import update_news, scheduler

app = Flask(__name__)
client = MongoClient("mongodb://<your-atlas-url>/")
db = client["crypto_news"]
collection = db["news"]

@app.route("/api/news", methods=["GET"])
def get_news():
    news = list(collection.find().sort("fetched_at", -1).limit(50))
    for item in news:
        item["_id"] = str(item["_id"])  # Convert ObjectId to string
    return jsonify(news)

if __name__ == "__main__":
    update_news()  # Initial fetch
    app.run(debug=True)
解释：
/api/news返回最新的 50 条新闻，按fetched_at排序。

import React, { useState, useEffect } from "react";
import "./NewsPage.css";

function NewsPage() {
    const [news, setNews] = useState([]);

    useEffect(() => {
        const fetchNews = async () => {
            try {
                const response = await fetch("http://localhost:5000/api/news");
                const data = await response.json();
                setNews(data);
            } catch (e) {
                console.error("Error:", e);
            }
        };
        fetchNews();
        const interval = setInterval(fetchNews, 300000);
        return () => clearInterval(interval);
    }, []);

    return (
        <div className="news-page">
            <h1>加密快讯</h1>
            {news.map(item => (
                <div key={item.link} className="news-item">
                    <span className="source">[{item.source}]</span>
                    <span className="title">
                        <a href={item.link} target="_blank" rel="noopener noreferrer">
                            {item.title}
                        </a>
                    </span>
                    <div className="time">{item.published}</div>
                </div>
            ))}
        </div>
    );
}

export default NewsPage;
